// Main App component
const App = () => {
  // --- General App State ---
  // Initialize wordInput with the DEFAULT_WORD_LIST_CONTENT
  const [wordInput, setWordInput] = useState(DEFAULT_WORD_LIST_CONTENT);
  const [wordList, setWordList] = useState([]);
  const [errorMessage, setErrorMessage] = useState('');
  const [successMessage, setSuccessMessage] = useState('');

  // --- Primary Search Criteria State ---
  const [newLetter, setNewLetter] = useState('');
  const [newPosition, setNewPosition] = useState('');
  const [newExclusivePosition, setNewExclusivePosition] = useState(true);
  const [searchCriteria, setSearchCriteria] = useState([]); // Array of {letter, positions[], exclusivePositions}
  const [wordLength, setWordLength] = useState(''); // Primary search word length
  const [excludedLetters, setExcludedLetters] = useState(''); // Primary search excluded letters

  // --- Search Mode State ---
  const [selectedSearchMode, setSelectedSearchMode] = useState('primary'); // 'primary', 'chained', 'patternMatch'

  // --- Chained Search State ---
  const [chainedSearchTargetPosition, setChainedSearchTargetPosition] = useState('');
  const [chainedSearchExclusiveTargetPosition, setChainedSearchExclusiveTargetPosition] = useState(true);
  const [chainedSearchWordLength, setChainedSearchWordLength] = useState(''); // Secondary word length
  const [secondaryWordMatchPosition, setSecondaryWordMatchPosition] = useState(''); // Position in secondary word

  // --- Pattern Match Search State ---
  const [patternSourceLength, setPatternSourceLength] = useState('');
  const [patternTargetLength, setPatternTargetLength] = useState('');
  const [newPatternSrcPos, setNewPatternSrcPos] = useState('');
  const [newPatternTargetPos, setNewPatternTargetPos] = useState('');
  const [patternMappings, setPatternMappings] = useState([]); // Array of {srcPos, targetPos}

  // --- Results State ---
  const [results, setResults] = useState([]); // Array of [primaryWord, secondaryWord] or [sourceWord, targetWord]

  // Effect to parse the word input whenever it changes
  useEffect(() => {
    const words = wordInput
      .split(/[\n,]+/)
      .filter(word => word.trim() !== '')
      .map(word => word.trim().toLowerCase());
    setWordList(words);
    // Clear success message when word input changes from user interaction
    if (wordInput !== DEFAULT_WORD_LIST_CONTENT) {
      setSuccessMessage('');
    }
  }, [wordInput]);

  // General Input Handlers
  const handleWordInputChange = (e) => { setWordInput(e.target.value); };
  const handleFileUpload = (e) => {
    setErrorMessage('');
    setSuccessMessage('');
    const file = e.target.files[0];
    if (file) {
      if (file.type !== 'text/csv') {
        setErrorMessage('Please upload a .csv file.');
        return;
      }
      const reader = new FileReader();
      reader.onload = (event) => {
        try {
          const content = event.target.result;
          setWordInput(content);
          setSuccessMessage(`Successfully loaded "${file.name}"!`);
        } catch (error) {
          setErrorMessage('Error reading file. Please ensure it is a valid CSV.');
          console.error('File reading error:', error);
        }
      };
      reader.onerror = () => { setErrorMessage('Failed to read file.'); };
      reader.readAsText(file);
    }
  };

  // Primary Criteria Handlers
  const handleAddCriterion = () => {
    if (!newLetter.trim()) { setErrorMessage('Please enter a letter.'); return; }
    if (newLetter.trim().length !== 1 || !/^[a-zA-Z]$/.test(newLetter.trim())) { setErrorMessage('Letter must be a single alphabet character.'); return; }
    if (!newPosition.trim()) { setErrorMessage('Please enter at least one position.'); return; }
    const positions = newPosition.trim().split(',').map(pos => parseInt(pos.trim(), 10));
    const invalidPositions = positions.filter(pos => isNaN(pos) || pos < 1);
    if (invalidPositions.length > 0) { setErrorMessage('All positions must be positive numbers separated by commas.'); return; }

    setSearchCriteria(prevCriteria => [
      ...prevCriteria,
      { letter: newLetter.trim().toLowerCase(), positions: positions, exclusivePositions: newExclusivePosition }
    ]);
    setNewLetter('');
    setNewPosition('');
    setErrorMessage('');
    setSuccessMessage('');
  };
  const handleRemoveCriterion = (indexToRemove) => {
    setSearchCriteria(prevCriteria => prevCriteria.filter((_, index) => index !== indexToRemove));
    setErrorMessage('');
    setSuccessMessage('');
  };

  // Pattern Match Mappings Handlers
  const handleAddPatternMapping = () => {
    const srcPos = parseInt(newPatternSrcPos.trim(), 10);
    const targetPos = parseInt(newPatternTargetPos.trim(), 10);

    if (isNaN(srcPos) || srcPos < 1 || isNaN(targetPos) || targetPos < 1) {
      setErrorMessage('Pattern mapping positions must be positive numbers.');
      return;
    }

    setPatternMappings(prevMappings => [...prevMappings, { srcPos, targetPos }]);
    setNewPatternSrcPos('');
    setNewPatternTargetPos('');
    setErrorMessage('');
  };

  const handleRemovePatternMapping = (indexToRemove) => {
    setPatternMappings(prevMappings => prevMappings.filter((_, index) => index !== indexToRemove));
  };


  // Clear All Function
  const handleClearAllCriteria = () => {
    setSearchCriteria([]);
    setNewLetter('');
    setNewPosition('');
    setNewExclusivePosition(true);
    setWordLength('');
    setExcludedLetters('');
    setSelectedSearchMode('primary'); // Reset search mode
    setChainedSearchTargetPosition('');
    setChainedSearchExclusiveTargetPosition(true);
    setChainedSearchWordLength('');
    setSecondaryWordMatchPosition('');
    setPatternSourceLength('');
    setPatternTargetLength('');
    setNewPatternSrcPos('');
    setNewPatternTargetPos('');
    setPatternMappings([]);
    setResults([]);
    setErrorMessage('');
    setSuccessMessage('All search criteria cleared.');
  };

  // Main Search Logic
  const handleSearch = () => {
    setErrorMessage('');
    setSuccessMessage('');
    setResults([]); // Clear previous results

    if (wordList.length === 0) {
      setErrorMessage('Please enter words or upload a CSV file first.');
      return;
    }

    // Parse explicitly excluded letters from primary search into a Set
    const parsedExcludedLetters = new Set(
      excludedLetters.toLowerCase().split('').filter(char => /^[a-z]$/.test(char))
    );

    // Define combined exclusions for secondary/pattern words (primary excluded + primary search criteria letters)
    // This correctly captures "all letters from the first search 'a', 'n' and 'b'" as described
    const combinedExclusionsForAdvancedSearches = new Set(parsedExcludedLetters);
    searchCriteria.forEach(criterion => {
      combinedExclusionsForAdvancedSearches.add(criterion.letter);
    });

    // Initialize finalCombinedResults here so it's always defined
    let finalCombinedResults = [];


    // --- Stage 1 Filtering (Primary Search - always runs first) ---
    const desiredPrimaryLength = parseInt(wordLength, 10);
    const filterByPrimaryLength = !isNaN(desiredPrimaryLength) && desiredPrimaryLength > 0;

    const initialFilteredWords = wordList.filter(word => {
      // 1. Check word length if specified
      if (filterByPrimaryLength && word.length !== desiredPrimaryLength) {
        return false;
      }

      // 2. Check for explicitly excluded letters (from primary input)
      for (const char of word) {
        if (parsedExcludedLetters.has(char)) {
          return false;
        }
      }

      // 3. Check all letter/position criteria (primary criteria)
      return searchCriteria.every(criterion => {
        const letter = criterion.letter;
        const specifiedPositionsSet = new Set(criterion.positions);

        const matchesAllSpecifiedPositions = criterion.positions.every(position => {
          return word[position - 1] === letter;
        });
        if (!matchesAllSpecifiedPositions) { return false; }

        if (criterion.exclusivePositions) {
          for (let i = 0; i < word.length; i++) {
            const currentPosition = i + 1;
            if (!specifiedPositionsSet.has(currentPosition) && word[i] === letter) {
              return false;
            }
          }
        }
        return true;
      });
    });

    // --- Advanced Search Modes ---
    if (selectedSearchMode === 'chained') {
      const targetPos = parseInt(chainedSearchTargetPosition, 10);
      const desiredSecondaryLength = parseInt(chainedSearchWordLength, 10);
      const secondaryMatchPos = parseInt(secondaryWordMatchPosition, 10);

      const isValidTargetPos = !isNaN(targetPos) && targetPos > 0;
      const isValidSecondaryLength = !isNaN(desiredSecondaryLength) && desiredSecondaryLength > 0;
      const isValidSecondaryMatchPos = !isNaN(secondaryMatchPos) && secondaryMatchPos > 0;

      if (!isValidTargetPos) { setErrorMessage('Chained: Please enter a valid positive number for primary word target position.'); return; }
      if (!isValidSecondaryLength) { setErrorMessage('Chained: Please enter a valid positive number for secondary word length.'); return; }
      if (!isValidSecondaryMatchPos) { setErrorMessage('Chained: Please enter a valid positive number for target letter position in secondary word.'); return; }

      const primaryWordsForChainedSearch = initialFilteredWords.filter(primaryWord => {
        if (primaryWord.length < targetPos) return false; // Primary word too short for target position

        if (chainedSearchExclusiveTargetPosition) {
          const letterAtTarget = primaryWord[targetPos - 1];
          if (!letterAtTarget) return false;
          for (let i = 0; i < primaryWord.length; i++) {
            if ((i + 1) !== targetPos && primaryWord[i] === letterAtTarget) {
              return false;
            }
          }
        }
        return true;
      });

      const secondarySearchMatchingLetters = new Set();
      primaryWordsForChainedSearch.forEach(word => {
        if (word.length >= targetPos) {
          secondarySearchMatchingLetters.add(word[targetPos - 1]);
        }
      });

      if (secondarySearchMatchingLetters.size === 0) {
        setErrorMessage('Chained: No primary words found (after strictness filter) with a letter at the specified chained search position.');
        return;
      }

      const secondaryMatches = wordList.filter(secondaryWord => {
        if (secondaryWord.length !== desiredSecondaryLength) { return false; }
        if (secondaryWord.length < secondaryMatchPos) { return false; } // Secondary word too short for match position

        for (const char of secondaryWord) {
          if (combinedExclusionsForAdvancedSearches.has(char)) { return false; }
        }
        if (!secondarySearchMatchingLetters.has(secondaryWord[secondaryMatchPos - 1])) { return false; }
        return true;
      });

      primaryWordsForChainedSearch.forEach(primaryWord => {
        const charAtTargetPos = primaryWord[targetPos - 1];
        secondaryMatches.forEach(secondaryWord => {
          if (secondaryWord[secondaryMatchPos - 1] === charAtTargetPos) {
            finalCombinedResults.push([primaryWord, secondaryWord]);
          }
        });
      });

      finalCombinedResults.sort((a, b) => {
        const secondaryComparison = a[1].localeCompare(b[1]);
        if (secondaryComparison !== 0) { return secondaryComparison; }
        return a[0].localeCompare(b[0]);
      });

    } else if (selectedSearchMode === 'patternMatch') {
      const srcLen = parseInt(patternSourceLength, 10);
      const targetLen = parseInt(patternTargetLength, 10);

      if (isNaN(srcLen) || srcLen < 1) { setErrorMessage('Pattern: Please enter a valid positive Source Word Length.'); return; }
      if (isNaN(targetLen) || targetLen < 1) { setErrorMessage('Pattern: Please enter a valid positive Target Word Length.'); return; }
      if (patternMappings.length === 0) { setErrorMessage('Pattern: Please add at least one character mapping.'); return; }

      // Validate pattern mappings against lengths
      for (const mapping of patternMappings) {
        if (mapping.srcPos > srcLen) {
          setErrorMessage(`Pattern: Mapping source position ${mapping.srcPos} must be <= Source Word Length ${srcLen}.`);
          return;
        }
        if (mapping.targetPos > targetLen) {
          setErrorMessage(`Pattern: Mapping target position ${mapping.targetPos} must be <= Target Word Length ${targetLen}.`);
          return;
        }
      }

      // Filter source and target words from the overall wordList (not primary results)
      // They are filtered by their specified lengths and the combined exclusions.
      const potentialSourceWords = wordList.filter(word => {
        if (word.length !== srcLen) return false;
        for (const char of word) { if (combinedExclusionsForAdvancedSearches.has(char)) return false; }
        return true;
      });

      const potentialTargetWords = wordList.filter(word => {
        if (word.length !== targetLen) return false;
        for (const char of word) { if (combinedExclusionsForAdvancedSearches.has(char)) return false; }
        return true;
      });

      // Perform the pattern matching
      potentialSourceWords.forEach(sourceWord => {
        potentialTargetWords.forEach(targetWord => {
          let matchesPattern = true;
          for (const mapping of patternMappings) {
            // Check if words are long enough for the current mapping's positions
            if (sourceWord.length < mapping.srcPos || targetWord.length < mapping.targetPos) {
              matchesPattern = false;
              break;
            }
            // Check if characters at specified positions match
            if (sourceWord[mapping.srcPos - 1] !== targetWord[mapping.targetPos - 1]) {
              matchesPattern = false;
              break;
            }
          }
          if (matchesPattern) {
            finalCombinedResults.push([sourceWord, targetWord]);
          }
        });
      });

      // Sort pattern match results by source word, then target word
      finalCombinedResults.sort((a, b) => {
        const sourceComparison = a[0].localeCompare(b[0]);
        if (sourceComparison !== 0) { return sourceComparison; }
        return a[1].localeCompare(b[1]);
      });

    } else {
      // Primary Search Only Mode
      // Ensure initialFilteredWords is mapped to the same structure as other results for consistent display
      finalCombinedResults = initialFilteredWords.map(word => [word]);
      finalCombinedResults.sort((a, b) => a[0].localeCompare(b[0])); // Sort primary results
    }

    setResults(finalCombinedResults);
  };

  return (
    <div className="min-h-screen bg-gradient-to-br from-blue-100 to-purple-100 p-4 font-sans flex items-center justify-center">
      <div className="bg-white p-8 rounded-xl shadow-2xl w-full max-w-4xl space-y-8">
        <h1 className="text-4xl font-extrabold text-center text-gray-800 mb-8">
          <span className="bg-clip-text text-transparent bg-gradient-to-r from-blue-600 to-purple-600">
            Word Searcher
          </span>
        </h1>

        {errorMessage && (
          <div className="bg-red-100 border border-red-400 text-red-700 px-4 py-3 rounded-md relative" role="alert">
            <strong className="font-bold">Error! </strong>
            <span className="block sm:inline">{errorMessage}</span>
          </div>
        )}

        {successMessage && (
          <div className="bg-green-100 border border-green-400 text-green-700 px-4 py-3 rounded-md relative" role="alert">
            <strong className="font-bold">Success! </strong>
            <span className="block sm:inline">{successMessage}</span>
          </div>
        )}

        {/* Word List Input Section */}
        <div className="space-y-4">
          <label htmlFor="word-list" className="block text-lg font-semibold text-gray-700">
            1. Enter Your Custom Word List (one word per line/comma-separated) OR Upload a CSV:
          </label>
          <div className="flex flex-col sm:flex-row gap-4 mb-4">
            <input
              type="file"
              id="csv-upload"
              accept=".csv"
              onChange={handleFileUpload}
              className="block w-full text-sm text-gray-500
                file:mr-4 file:py-2 file:px-4
                file:rounded-full file:border-0
                file:text-sm file:font-semibold
                file:bg-purple-50 file:text-purple-700
                hover:file:bg-purple-100
                cursor-pointer"
            />
          </div>
          <textarea
            id="word-list"
            className="w-full p-4 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-400 focus:border-transparent transition duration-200 ease-in-out resize-y min-h-[150px] text-gray-800"
            rows="10"
            value={wordInput}
            onChange={handleWordInputChange}
            placeholder="e.g.,&#x0A;apple&#x0A;banana&#x0A;grape&#x0A;orange&#x0A;apricot&#x0A;or paste CSV content here..."
          ></textarea>
          <p className="text-sm text-gray-500">
            Currently loaded words: <span className="font-medium text-blue-600">{wordList.length}</span>
          </p>
        </div>

        {/* Primary Search Criteria Section */}
        <div className="space-y-4">
          <h2 className="text-lg font-semibold text-gray-700">
            2. Primary Search Criteria:
          </h2>
          <div className="flex flex-col sm:flex-row gap-4">
            <input
              type="text"
              className="flex-1 p-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-400 focus:border-transparent transition duration-200 ease-in-out text-gray-800"
              placeholder="Letter (e.g., 'a')"
              value={newLetter}
              onChange={(e) => setNewLetter(e.target.value)}
              maxLength="1"
            />
            <input
              type="text"
              className="flex-1 p-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-400 focus:border-transparent transition duration-200 ease-in-out text-gray-800"
              placeholder="Positions (e.g., '1,3,5')"
              value={newPosition}
              onChange={(e) => setNewPosition(e.target.value)}
            />
            <button
              onClick={handleAddCriterion}
              className="px-6 py-3 bg-blue-500 text-white font-bold rounded-lg shadow-md hover:bg-blue-600 focus:outline-none focus:ring-2 focus:ring-blue-400 focus:ring-offset-2 transition duration-200 ease-in-out transform hover:scale-105"
            >
              Add Criterion
            </button>
          </div>

          {/* Exclusive Position Checkbox */}
          <div className="flex items-center mt-2">
            <input
              type="checkbox"
              id="exclusive-position"
              checked={newExclusivePosition}
              onChange={(e) => setNewExclusivePosition(e.target.checked)}
              className="h-5 w-5 text-blue-600 rounded border-gray-300 focus:ring-blue-500"
            />
            <label htmlFor="exclusive-position" className="ml-2 text-gray-700">
              Letter only at specified positions (strict match)
            </label>
          </div>

          {/* Word Length Input */}
          <div className="flex flex-col sm:flex-row gap-4 mb-4 mt-4">
            <input
              type="number"
              className="flex-1 p-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-400 focus:border-transparent transition duration-200 ease-in-out text-gray-800"
              placeholder="Exact Word Length (optional)"
              value={wordLength}
              onChange={(e) => setWordLength(e.target.value)}
              min="1"
            />
          </div>

          {/* Excluded Letters Input */}
          <div className="flex flex-col sm:flex-row gap-4">
            <input
              type="text"
              className="flex-1 p-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-400 focus:border-transparent transition duration-200 ease-in-out text-gray-800"
              placeholder="Excluded Letters (e.g., 'snx')"
              value={excludedLetters}
              onChange={(e) => setExcludedLetters(e.target.value)}
            />
          </div>

          {/* Display Current Primary Criteria */}
          {searchCriteria.length > 0 && (
            <div className="mt-4 p-4 bg-blue-50 rounded-lg border border-blue-200">
              <h3 className="text-md font-semibold text-blue-800 mb-2">Current Letter/Position Criteria:</h3>
              <ul className="list-disc list-inside space-y-2">
                {searchCriteria.map((criterion, index) => (
                  <li key={index} className="flex items-center justify-between text-gray-700">
                    <span>
                      Letter: '<span className="font-bold text-blue-700">{criterion.letter}</span>' at Position(s): <span className="font-bold text-blue-700">{criterion.positions.join(', ')}</span>
                      {criterion.exclusivePositions ? ' (strict)' : ' (flexible)'}
                    </span>
                    <button
                      onClick={() => handleRemoveCriterion(index)}
                      className="ml-4 text-red-500 hover:text-red-700 focus:outline-none"
                      title="Remove criterion"
                    >
                      <svg xmlns="http://www.w3.org/2000/svg" className="h-5 w-5" viewBox="0 0 20 20" fill="currentColor">
                        <path fillRule="evenodd" d="M9 2a1 1 0 00-.894.553L7.382 4H4a1 1 0 000 2v10a2 2 0 002 2h8a2 2 0 002-2V6a1 1 0 100-2h-3.382l-.724-1.447A1 1 0 0011 2H9zM7 8a1 1 0 012 0v6a1 1 0 11-2 0V8zm6 0a1 1 0 11-2 0v6a1 1 0 112 0V8z" clipRule="evenodd" />
                      </svg>
                    </button>
                  </li>
                ))}
              </ul>
              <div className="mt-4 text-right">
                <button
                  onClick={handleClearAllCriteria}
                  className="px-4 py-2 bg-gray-200 text-gray-700 font-semibold rounded-lg shadow-md hover:bg-gray-300 focus:outline-none focus:ring-2 focus:ring-gray-400 focus:ring-offset-2 transition duration-200 ease-in-out"
                >
                  Clear All Criteria
                </button>
              </div>
            </div>
          )}
        </div>

        {/* Search Mode Selector */}
        <div className="space-y-4 p-4 bg-gray-50 rounded-xl border border-gray-200">
          <h2 className="text-lg font-semibold text-gray-800">
            3. Select Search Mode:
          </h2>
          <div className="flex flex-col sm:flex-row gap-4">
            <label className="inline-flex items-center">
              <input
                type="radio"
                className="form-radio h-5 w-5 text-blue-600"
                name="searchMode"
                value="primary"
                checked={selectedSearchMode === 'primary'}
                onChange={(e) => setSelectedSearchMode(e.target.value)}
              />
              <span className="ml-2 text-gray-700">Primary Search Only</span>
            </label>
            <label className="inline-flex items-center">
              <input
                type="radio"
                className="form-radio h-5 w-5 text-purple-600"
                name="searchMode"
                value="chained"
                checked={selectedSearchMode === 'chained'}
                onChange={(e) => setSelectedSearchMode(e.target.value)}
              />
              <span className="ml-2 text-gray-700">Chained Search</span>
            </label>
            <label className="inline-flex items-center">
              <input
                type="radio"
                className="form-radio h-5 w-5 text-green-600"
                name="searchMode"
                value="patternMatch"
                checked={selectedSearchMode === 'patternMatch'}
                onChange={(e) => setSelectedSearchMode(e.target.value)}
              />
              <span className="ml-2 text-gray-700">Pattern Match Search</span>
            </label>
          </div>
        </div>


        {/* Chained Search Section (Conditional Rendering) */}
        {selectedSearchMode === 'chained' && (
          <div className="space-y-4 p-4 bg-purple-50 rounded-xl border border-purple-200">
            <h2 className="text-lg font-semibold text-purple-800">
              Chained Search Criteria:
            </h2>
            <p className="text-sm text-gray-600">
              Filters primary results. The word's letter at the specified position must be the matching letter of a secondary word from your list. Secondary words are filtered by length and cannot contain letters from primary excluded list or primary search criteria.
            </p>
            <input
              type="number"
              className="w-full p-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-purple-400 focus:border-transparent transition duration-200 ease-in-out text-gray-800"
              placeholder="Target Position in Primary Word (e.g., '2')"
              value={chainedSearchTargetPosition}
              onChange={(e) => setChainedSearchTargetPosition(e.target.value)}
              min="1"
            />
            <div className="flex items-center mt-2">
              <input
                type="checkbox"
                id="chained-exclusive-position"
                checked={chainedSearchExclusiveTargetPosition}
                onChange={(e) => setChainedSearchExclusiveTargetPosition(e.target.checked)}
                className="h-5 w-5 text-purple-600 rounded border-gray-300 focus:ring-purple-500"
              />
              <label htmlFor="chained-exclusive-position" className="ml-2 text-gray-700">
                Target letter only at specified position in primary word (strict)
              </label>
            </div>

            <input
              type="number"
              className="w-full p-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-purple-400 focus:border-transparent transition duration-200 ease-in-out text-gray-800"
              placeholder="Secondary Word Length (e.g., '2')"
              value={chainedSearchWordLength}
              onChange={(e) => setChainedSearchWordLength(e.target.value)}
              min="1"
            />
            <input
              type="number"
              className="w-full p-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-purple-400 focus:border-transparent transition duration-200 ease-in-out text-gray-800"
              placeholder="Target Letter Position in Secondary Word (e.g., '1')"
              value={secondaryWordMatchPosition}
              onChange={(e) => setSecondaryWordMatchPosition(e.target.value)}
              min="1"
            />
          </div>
        )}

        {/* Pattern Match Search Section (Conditional Rendering) */}
        {selectedSearchMode === 'patternMatch' && (
          <div className="space-y-4 p-4 bg-green-50 rounded-xl border border-green-200">
            <h2 className="text-lg font-semibold text-green-800">
              Pattern Match Search Criteria:
            </h2>
            <p className="text-sm text-gray-600">
              Finds pairs of words (Source {'->'} Target) from your list where characters match at specified positions. Both words in the pair must also adhere to **Excluded Letters** and **Primary Criteria letters** defined above. You do not specify the letters, only their positions.
            </p>
            <div className="flex flex-col sm:flex-row gap-4">
              <input
                type="number"
                className="flex-1 p-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-green-400 focus:border-transparent transition duration-200 ease-in-out text-gray-800"
                placeholder="Source Word Length (e.g., '3')"
                value={patternSourceLength}
                onChange={(e) => setPatternSourceLength(e.target.value)}
                min="1"
              />
              <input
                type="number"
                className="flex-1 p-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-green-400 focus:border-transparent transition duration-200 ease-in-out text-gray-800"
                placeholder="Target Word Length (e.g., '2')"
                value={patternTargetLength}
                onChange={(e) => setPatternTargetLength(e.target.value)}
                min="1"
              />
            </div>
            <h3 className="text-md font-semibold text-green-700 mt-4">Add Character Mapping (Source Position {'->'} Target Position):</h3>
            <p className="text-sm text-gray-600">
              Specify which character position in the source word should match which character position in the target word.
            </p>
            <div className="flex flex-col sm:flex-row gap-4">
              <input
                type="number"
                className="flex-1 p-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-green-400 focus:border-transparent transition duration-200 ease-in-out text-gray-800"
                placeholder="Source Pos (e.g., '1')"
                value={newPatternSrcPos}
                onChange={(e) => setNewPatternSrcPos(e.target.value)}
                min="1"
              />
              <input
                type="number"
                className="flex-1 p-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-green-400 focus:border-transparent transition duration-200 ease-in-out text-gray-800"
                placeholder="Target Pos (e.g., '1')"
                value={newPatternTargetPos}
                onChange={(e) => setNewPatternTargetPos(e.target.value)}
                min="1"
              />
              <button
                onClick={handleAddPatternMapping}
                className="px-6 py-3 bg-green-500 text-white font-bold rounded-lg shadow-md hover:bg-green-600 focus:outline-none focus:ring-2 focus:ring-green-400 focus:ring-offset-2 transition duration-200 ease-in-out transform hover:scale-105"
              >
                Add Mapping
              </button>
            </div>
            {patternMappings.length > 0 && (
              <div className="mt-4 p-3 bg-green-100 rounded-lg border border-green-300">
                <h4 className="text-sm font-semibold text-green-800 mb-2">Current Mappings:</h4>
                <ul className="list-disc list-inside space-y-1">
                  {patternMappings.map((map, index) => (
                    <li key={index} className="flex items-center justify-between text-gray-700">
                      <span>
                        Source Pos: <span className="font-bold text-green-700">{map.srcPos}</span> {'->'} Target Pos: <span className="font-bold text-green-700">{map.targetPos}</span>
                      </span>
                      <button
                        onClick={() => handleRemovePatternMapping(index)}
                        className="ml-4 text-red-500 hover:text-red-700 focus:outline-none"
                        title="Remove mapping"
                      >
                        <svg xmlns="http://www.w3.org/2000/svg" className="h-5 w-5" viewBox="0 0 20 20" fill="currentColor">
                          <path fillRule="evenodd" d="M9 2a1 1 0 00-.894.553L7.382 4H4a1 1 0 000 2v10a2 2 0 002 2h8a2 2 0 002-2V6a1 1 0 100-2h-3.382l-.724-1.447A1 1 0 0011 2H9zM7 8a1 1 0 012 0v6a1 1 0 11-2 0V8zm6 0a1 1 0 11-2 0v6a1 1 0 112 0V8z" clipRule="evenodd" />
                        </svg>
                      </button>
                    </li>
                  ))}
                </ul>
              </div>
            )}
          </div>
        )}

        {/* Search Button */}
        <div className="text-center">
          <button
            onClick={handleSearch}
            className="px-8 py-4 bg-purple-600 text-white font-extrabold text-xl rounded-lg shadow-xl hover:bg-purple-700 focus:outline-none focus:ring-4 focus:ring-purple-400 focus:ring-offset-2 transition duration-300 ease-in-out transform hover:scale-105"
          >
            4. Perform Search
          </button>
        </div>

        {/* Results Section */}
        <div className="space-y-4">
          <h2 className="text-2xl font-bold text-gray-800">
            Search Results:
          </h2>
          {results.length > 0 ? (
            <div className="p-4 bg-green-50 rounded-lg border border-green-200">
              <p className="text-lg text-green-800 mb-2">
                Found <span className="font-bold">{results.length}</span> matching pair(s):
              </p>
              <ul className="list-disc list-inside grid grid-cols-1 sm:grid-cols-2 md:grid-cols-3 gap-2">
                {results.map((pair, index) => (
                  <li key={index} className="text-gray-700 text-lg font-medium">
                    {pair.length > 1 ? `${pair[0]} -> ${pair[1]}` : pair[0]}
                  </li>
                ))}
              </ul>
            </div>
          ) : (
            <p className="text-gray-600 text-lg p-4 bg-gray-50 rounded-lg border border-gray-200">
              No words found matching the criteria, or search not yet performed.
            </p>
          )}
        </div>
      </div>
    </div>
  );
};

export default App;
